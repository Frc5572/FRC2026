cmake_minimum_required(VERSION 3.15)
project(Whacknet)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

find_package(Eigen3 REQUIRED)
find_package(GTSAM REQUIRED)

option(BUILD_SHARED_LIBS "Build Whacknet as shared library" OFF)

add_library(whacknet STATIC src/whacknet.cc)
target_include_directories(whacknet PUBLIC 
    ${CMAKE_CURRENT_SOURCE_DIR}
    ${GTSAM_INCLUDE_DIR}
)
target_link_libraries(whacknet PUBLIC 
    gtsam
    Eigen3::Eigen
    pthread
)
target_compile_options(whacknet PRIVATE -O3 -march=native)

add_library(robot_localizer STATIC src/robot_localizer.cc)
target_include_directories(robot_localizer PUBLIC 
    ${CMAKE_CURRENT_SOURCE_DIR}
    ${GTSAM_INCLUDE_DIR}
)
target_link_libraries(robot_localizer PUBLIC 
    whacknet
    gtsam
    Eigen3::Eigen
    pthread
)
target_compile_options(robot_localizer PRIVATE -O3 -march=native)
